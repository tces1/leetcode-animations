84. 柱状图中最大的矩形 · 单调栈动画

栈里存下标,高度单调递增。遇到更矮的柱子时,以栈顶柱子为「高」,弹出并向两侧扩展算面积:面积 = 高 × 宽,其中宽 = 右边界 − 左边界 − 1。

// Java · 单调递增栈 · 时间/空间 O(N)
class Solution {
    public int largestRectangleArea(int[] h) {
        int n = h.length, max = 0;
        Deque<Integer> stack = new ArrayDeque<>();
        for (int i = 0; i <= n; i++) {
            int cur = (i == n) ? 0 : h[i];       // 末尾哨兵高度 0，清空栈
            while (!stack.isEmpty() && cur < h[stack.peek()]) {
                int height = h[stack.pop()];     // 以它为矩形的高
                int left = stack.isEmpty() ? -1 : stack.peek();
                int width = i - left - 1;        // 左右两个「更矮」边界之间
                max = Math.max(max, height * width);
            }
            stack.push(i);
        }
        return max;
    }
}
# Python
class Solution:
    def largestRectangleArea(self, h: List[int]) -> int:
        stack, mx, n = [], 0, len(h)
        for i in range(n + 1):
            cur = 0 if i == n else h[i]
            while stack and cur < h[stack[-1]]:
                height = h[stack.pop()]
                left = stack[-1] if stack else -1
                mx = max(mx, height * (i - left - 1))
            stack.append(i)
        return mx
