70. 爬楼梯 · 从递归到滚动 DP 的三步优化

每次可以爬 1 或 2 个台阶，问爬到第 n 阶有多少种不同方法。答案满足 f(n) = f(n-1) + f(n-2)，基例 f(1)=1、f(2)=2。同一道题用三种解法逐帧对比：朴素递归（大量重复子问题）→ 记忆化搜索（缓存去重，O(n)）→ 滚动 DP（只留两个变量，空间 O(1)）。

// Java · 三种解法，基例与递推式完全一致
class Solution {
    // ① 朴素递归 —— 时间 O(2ⁿ)，重复子问题爆炸
    public int climbRecursive(int n) {
        if (n <= 2) return n;
        return climbRecursive(n - 1) + climbRecursive(n - 2);
    }

    // ② 记忆化搜索 —— 缓存去重，时间 O(n)、空间 O(n)
    public int climbMemo(int n) {
        int[] memo = new int[n + 1];
        return dfs(n, memo);
    }
    private int dfs(int n, int[] memo) {
        if (n <= 2) return n;
        if (memo[n] != 0) return memo[n];       // 命中缓存直接返回
        return memo[n] = dfs(n - 1, memo) + dfs(n - 2, memo);
    }

    // ③ 滚动 DP —— 只留前两个状态，空间 O(1)
    public int climbStairs(int n) {
        if (n <= 2) return n;
        int prev = 1, cur = 2;                   // f(1)=1, f(2)=2
        for (int i = 3; i <= n; i++) {
            int next = prev + cur;               // f(i)=f(i-1)+f(i-2)
            prev = cur;
            cur = next;                          // 窗口右移一格
        }
        return cur;
    }
}
# Python · 三种解法
class Solution:
    # ① 朴素递归
    def climb_recursive(self, n: int) -> int:
        if n <= 2:
            return n
        return self.climb_recursive(n - 1) + self.climb_recursive(n - 2)

    # ② 记忆化搜索
    def climb_memo(self, n: int) -> int:
        from functools import lru_cache
        @lru_cache(maxsize=None)
        def dfs(k: int) -> int:
            if k <= 2:
                return k
            return dfs(k - 1) + dfs(k - 2)       # 相同 k 只算一次
        return dfs(n)

    # ③ 滚动 DP · 空间 O(1)
    def climb_stairs(self, n: int) -> int:
        if n <= 2:
            return n
        prev, cur = 1, 2                         # f(1), f(2)
        for _ in range(3, n + 1):
            prev, cur = cur, prev + cur          # 窗口右移一格
        return cur
