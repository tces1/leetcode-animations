// Java · 二维 DP(区间) · 时间/空间 O(n²)
class Solution {
    public String longestPalindrome(String s) {
        int n = s.length();
        boolean[][] dp = new boolean[n][n];
        int start = 0, maxLen = 1;
        for (int i = 0; i < n; i++) dp[i][i] = true;   // 长度 1
        for (int L = 2; L <= n; L++) {                  // 按长度从小到大
            for (int i = 0; i + L - 1 < n; i++) {
                int j = i + L - 1;
                if (s.charAt(i) != s.charAt(j)) dp[i][j] = false;
                else if (L == 2) dp[i][j] = true;         // 两端相等且长度 2
                else dp[i][j] = dp[i + 1][j - 1];         // 看里面那格
                if (dp[i][j] && L > maxLen) { start = i; maxLen = L; }
            }
        }
        return s.substring(start, start + maxLen);
    }
}
# Python · 二维 DP(区间) · 时间/空间 O(n²)
class Solution:
    def longestPalindrome(self, s: str) -> str:
        n = len(s)
        dp = [[False] * n for _ in range(n)]
        start, maxLen = 0, 1
        for i in range(n):
            dp[i][i] = True                 # 长度 1
        for L in range(2, n + 1):           # 按长度从小到大
            for i in range(n - L + 1):
                j = i + L - 1
                if s[i] != s[j]:
                    dp[i][j] = False
                elif L == 2:
                    dp[i][j] = True         # 两端相等且长度 2
                else:
                    dp[i][j] = dp[i + 1][j - 1]   # 看里面那格
                if dp[i][j] and L > maxLen:
                    start, maxLen = i, L
        return s[start:start + maxLen]
