// Java · 单调递减栈 · 时间/空间 O(N)
class Solution {
    public int[] dailyTemperatures(int[] t) {
        int n = t.length;
        int[] ans = new int[n];              // 默认全 0
        Deque<Integer> stack = new ArrayDeque<>();  // 存下标
        for (int i = 0; i < n; i++) {
            while (!stack.isEmpty() && t[i] > t[stack.peek()]) {
                int j = stack.pop();
                ans[j] = i - j;              // 第 j 天等到了更暖的第 i 天
            }
            stack.push(i);
        }
        return ans;
    }
}
# Python
class Solution:
    def dailyTemperatures(self, t: List[int]) -> List[int]:
        ans = [0] * len(t)
        stack = []                       # 存下标
        for i, cur in enumerate(t):
            while stack and cur > t[stack[-1]]:
                j = stack.pop()
                ans[j] = i - j
            stack.append(i)
        return ans
