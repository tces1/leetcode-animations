// Java · 记忆化搜索(自顶向下 DP) · 时间 O(n√n) 空间 O(n)
class Solution {
    private static int[] memo = new int[0];      // 跨用例共享,避免重复计算

    public int numSquares(int n) {
        if (memo.length < n + 1) memo = new int[n + 1];  // 0 表示尚未计算
        return dfs(n);
    }

    private int dfs(int i) {
        if (i < 1) return 0;
        if (memo[i] != 0) return memo[i];        // 命中缓存
        int step = Integer.MAX_VALUE;
        for (int j = 1; j * j <= i; j++)          // 枚举最后减去的那块 j²
            step = Math.min(step, dfs(i - j * j) + 1);
        return memo[i] = step;
    }
}
# Python · 记忆化搜索 · @cache 放到类外,多个用例共享缓存
from functools import cache
from math import inf

@cache
def dfs(i):
    if i < 1:
        return 0
    j = 0
    step = inf
    while True:
        j += 1
        if j * j > i:
            break
        step = min(step, dfs(i - j * j) + 1)   # 最后一块用 j²
    return step

class Solution:
    def numSquares(self, n: int) -> int:
        return dfs(n)
