763. 划分字母区间 · 贪心边界动画

把字符串切成尽量多的片段，同一个字母只能出现在一个片段里。先记下每个字母最后出现的下标；再从左扫描，用 end 记录「当前片段必须延伸到的最右位置」（取片段内所有字母 last 的最大值），一旦 i == end 就地切一刀。时间 O(n)、空间 O(1)（字符集 ≤ 26）。

// Java · 贪心 · 时间 O(n)、空间 O(1)
class Solution {
    public List<Integer> partitionLabels(String s) {
        int[] last = new int[26];
        for (int i = 0; i < s.length(); i++)          // 记录每个字母最后出现的下标
            last[s.charAt(i) - 'a'] = i;
        List<Integer> res = new ArrayList<>();
        int start = 0, end = 0;
        for (int i = 0; i < s.length(); i++) {
            end = Math.max(end, last[s.charAt(i) - 'a']); // 片段必须延伸到这里
            if (i == end) {                            // 片段内所有字母都收尾了 → 切一刀
                res.add(end - start + 1);
                start = i + 1;
            }
        }
        return res;
    }
}
# Python
class Solution:
    def partitionLabels(self, s: str) -> List[int]:
        last = {c: i for i, c in enumerate(s)}   # 每个字母最后出现的下标
        res = []
        start = end = 0
        for i, c in enumerate(s):
            end = max(end, last[c])              # 片段必须延伸到这里
            if i == end:                         # 收尾 → 切一刀
                res.append(end - start + 1)
                start = i + 1
        return res
