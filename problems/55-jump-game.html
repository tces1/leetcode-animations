55. 跳跃游戏 · 贪心「最远可达」动画

从下标 0 出发，每个位置 nums[i] 表示最多能往右跳几步。只需从左到右扫一遍，维护「目前能到达的最远下标 maxReach」：走到位置 i 时若 i > maxReach 说明这里根本走不到，直接失败；否则用 i + nums[i] 刷新 maxReach。只要 maxReach 覆盖到最后一格就成功。时间 O(n)、空间 O(1)。

// Java · 贪心一次遍历 · 时间 O(n)、空间 O(1)
class Solution {
    public boolean canJump(int[] nums) {
        int maxReach = 0;                       // 目前能到达的最远下标
        for (int i = 0; i < nums.length; i++) {
            if (i > maxReach) return false;     // i 都到不了 → 被卡住
            maxReach = Math.max(maxReach, i + nums[i]); // 用 i 更新最远可达
        }
        return true;                            // 走完全程都没卡住
    }
}
# Python
class Solution:
    def canJump(self, nums: List[int]) -> bool:
        max_reach = 0                           # 目前能到达的最远下标
        for i, step in enumerate(nums):
            if i > max_reach:                    # i 都到不了 → 被卡住
                return False
            max_reach = max(max_reach, i + step) # 用 i 更新最远可达
        return True                             # 走完全程都没卡住
