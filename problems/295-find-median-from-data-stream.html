# Python 3.14+：heapq 提供了公开的大顶堆版本函数
from heapq import heappush, heappushpop, heappush_max, heappushpop_max

class MedianFinder:
    def __init__(self):
        self.left = []   # 大顶堆：较小的一半
        self.right = []  # 小顶堆：较大的一半

    def addNum(self, num: int) -> None:
        if len(self.left) == len(self.right):
            # 让 num 经 right 中转，把「该进小半区的数」放进 left
            heappush_max(self.left, heappushpop(self.right, num))
        else:
            # 让 num 经 left 中转，把「该进大半区的数」放进 right
            heappush(self.right, heappushpop_max(self.left, num))

    def findMedian(self) -> float:
        if len(self.left) > len(self.right):
            return self.left[0]
        return (self.left[0] + self.right[0]) / 2

# 若 Python < 3.14 没有 *_max：把值取负存入普通小顶堆即可模拟大顶堆
// Java · 两个优先队列 · addNum O(log n)，findMedian O(1)
class MedianFinder {
    // left 大顶堆：较小的一半；right 小顶堆：较大的一半
    private final PriorityQueue<Integer> left = new PriorityQueue<>(Collections.reverseOrder());
    private final PriorityQueue<Integer> right = new PriorityQueue<>();

    public void addNum(int num) {
        if (left.size() == right.size()) {
            right.offer(num);
            left.offer(right.poll());   // = heappushpop(right, num) 后进 left
        } else {
            left.offer(num);
            right.offer(left.poll());   // = heappushpop_max(left, num) 后进 right
        }
    }

    public double findMedian() {
        if (left.size() > right.size()) return left.peek();
        return (left.peek() + right.peek()) / 2.0;
    }
}
