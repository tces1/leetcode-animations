// Java · 一次遍历 · 时间 O(n)、空间 O(1)
class Solution {
    public int maxProfit(int[] prices) {
        int minPrice = Integer.MAX_VALUE, maxProfit = 0;
        for (int p : prices) {
            maxProfit = Math.max(maxProfit, p - minPrice); // 今天卖出能赚多少
            minPrice = Math.min(minPrice, p);              // 刷新最低买入价
        }
        return maxProfit;
    }
}
# Python
class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        min_price = float('inf')
        max_profit = 0
        for p in prices:
            max_profit = max(max_profit, p - min_price)  # 今天卖出能赚多少
            min_price = min(min_price, p)                # 刷新最低买入价
        return max_profit
