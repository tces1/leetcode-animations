45. 跳跃游戏 II · 贪心分层动画

保证能到终点，求最少跳几次。把每一跳看成一个「层」：在当前跳能落到的窗口 [i .. curEnd] 内边走边记录下一跳最远能到哪（farthest）；一旦走到窗口边界 curEnd，就必须再跳一次，并把边界推到 farthest。时间 O(n)、空间 O(1)。

// Java · 贪心分层(隐式 BFS) · 时间 O(n)、空间 O(1)
class Solution {
    public int jump(int[] nums) {
        int jumps = 0, curEnd = 0, farthest = 0;
        for (int i = 0; i < nums.length - 1; i++) {   // 注意到 n-2 即可
            farthest = Math.max(farthest, i + nums[i]); // 本跳窗口内，下一跳最远能到哪
            if (i == curEnd) {                          // 走到当前窗口边界，必须再跳一次
                jumps++;
                curEnd = farthest;                      // 边界推进到下一层
            }
        }
        return jumps;
    }
}
# Python
class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        jumps = cur_end = farthest = 0
        for i in range(n - 1):                    # 到 n-2 即可
            farthest = max(farthest, i + nums[i]) # 下一跳最远能到哪
            if i == cur_end:                      # 走到窗口边界，必须再跳
                cur_end = farthest                # 边界推进到下一层
                jumps += 1
        return jumps
