322. 零钱兑换 · 动态规划填表动画

dp[i] = 凑出金额 i 所需最少的硬币数。填 dp[i] 时，枚举「最后一枚硬币 c」（需 c ≤ i），答案就是 min(dp[i − c]) + 1。凑不出则记为 ∞，最终 ∞ 返回 −1。

// Java · 自底向上填表 · 时间 O(amount × coins) 空间 O(amount)
class Solution {
    public int coinChange(int[] coins, int amount) {
        int[] dp = new int[amount + 1];
        Arrays.fill(dp, amount + 1);         // amount+1 当作 ∞(硬币数不可能超过 amount)
        dp[0] = 0;                           // 凑出 0 需要 0 枚
        for (int i = 1; i <= amount; i++)
            for (int c : coins)              // 枚举最后一枚硬币 c
                if (c <= i)
                    dp[i] = Math.min(dp[i], dp[i - c] + 1);
        return dp[amount] > amount ? -1 : dp[amount];
    }
}
# Python · 记忆化搜索 · @cache 必须放在方法内(coins 因用例而异)
class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        @cache
        def dfs(i):
            if i == 0:
                return 0
            step = inf
            for c in coins:
                if i - c >= 0:               # 提前剪枝,不递归到负数
                    step = min(step, dfs(i - c) + 1)
            return step
        res = dfs(amount)
        return -1 if res == inf else res
# Python · 自底向上填表(与本页动画一致,无递归、不会爆栈)
class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        dp = [0] + [inf] * amount           # dp[i] = 凑出 i 的最少硬币数
        for i in range(1, amount + 1):
            for c in coins:
                if c <= i:                   # 最后一枚用 c
                    dp[i] = min(dp[i], dp[i - c] + 1)
        return dp[amount] if dp[amount] != inf else -1
